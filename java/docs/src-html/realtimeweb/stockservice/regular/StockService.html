001    package realtimeweb.stockservice.regular;
002    
003    import java.util.ArrayList;
004    
005    import realtimeweb.stockservice.domain.Stock;
006    import realtimeweb.stockservice.json.JsonGetStockInformationListener;
007    import realtimeweb.stockservice.json.JsonStockService;
008    import realtimeweb.stockservice.main.AbstractStockService;
009    
010    import com.google.gson.Gson;
011    import com.google.gson.JsonArray;
012    import com.google.gson.JsonParser;
013    
014    /**
015     * Used to get data as classes.
016     */
017    public class StockService implements AbstractStockService {
018            private static StockService instance;
019            private JsonStockService jsonInstance;
020            private Gson gson;
021            /**
022             * **For internal use only!** Protected Constructor guards against instantiation.
023            
024             * @return 
025             */
026            protected  StockService() {
027                    this.jsonInstance = JsonStockService.getInstance();
028                    this.gson = new Gson();
029            }
030            
031            /**
032             * Retrieves the singleton instance.
033            
034             * @return StockService
035             */
036            public static StockService getInstance() {
037                    if (instance == null) {
038                            synchronized (StockService.class) {
039                                    if (instance == null) {
040                                            instance = new StockService();
041                                    }
042                            }
043                            
044                    }
045                    return instance;
046            }
047            
048            /**
049             * Establishes a connection to the online service. Requires an internet connection.
050            
051             */
052            @Override
053            public void connect() {
054                    jsonInstance.connect();
055            }
056            
057            /**
058             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
059            
060             */
061            @Override
062            public void disconnect() {
063                    jsonInstance.disconnect();
064            }
065            
066            /**
067             * Retrieves current stock information.
068             * @param ticker A comma separated list of ticker symbols (e.g. "AAPL, MSFT, CSCO").
069             * @return ArrayList<Stock>
070             */
071            public ArrayList<Stock> getStockInformation(String ticker) throws Exception {
072                    String response = jsonInstance.getStockInformation(ticker).substring(3);
073                    JsonParser parser = new JsonParser();
074                    JsonArray allChildren = parser.parse(response).getAsJsonArray();
075                    ArrayList<Stock> result = new ArrayList<Stock>();
076                    for (int i = 0; i < allChildren.size(); i += 1) {
077                            result.add(new Stock(allChildren.get(i).getAsJsonObject(), gson));
078                    }
079                    return result;
080            }
081            
082            /**
083             * Retrieves current stock information.
084             * @param ticker A comma separated list of ticker symbols (e.g. "AAPL, MSFT, CSCO").
085             * @param callback The listener that will receive the data (or error).
086             */
087            public void getStockInformation(String ticker, final GetStockInformationListener callback) {
088                    
089                    jsonInstance.getStockInformation(ticker, new JsonGetStockInformationListener() {
090                        @Override
091                        public void getStockInformationFailed(Exception exception) {
092                            callback.getStockInformationFailed(exception);
093                        }
094                        
095                        @Override
096                        public void getStockInformationCompleted(String response) {
097                            JsonParser parser = new JsonParser();
098                          JsonArray allChildren = parser.parse(response.substring(3)).getAsJsonArray();
099                            ArrayList<Stock> result = new ArrayList<Stock>();
100                            for (int i = 0; i < allChildren.size(); i += 1) {
101                                result.add(new Stock(allChildren.get(i).getAsJsonObject(), gson));
102                            }
103                            callback.getStockInformationCompleted(result);
104                        }
105                    });
106                    
107            }
108            
109    }
