001    package realtimeweb.stockservice.structured;
002    
003    import java.util.ArrayList;
004    /**
005     * A listener for the getStockInformation method. On success, passes the data into the getStockInformationCompleted method. On failure, passes the exception to the getStockInformationFailed method.
006     */
007    public interface StructuredGetStockInformationListener {
008            /**
009             * 
010             * @param arrayList The method that should be overridden to handle the data if the method was successful.
011             */
012            public abstract void getStockInformationCompleted(ArrayList arrayList);
013            /**
014             * 
015             * @param error The method that should be overridden to handle an exception that occurred while getting the SearchResponse.
016             */
017            public abstract void getStockInformationFailed(Exception error);
018    }
