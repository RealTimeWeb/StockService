001    package realtimeweb.stockservice.json;
002    
003    /**
004     * A listener for the getStockInformation method. On success, passes the data into the getStockInformationCompleted method. On failure, passes the exception to the getStockInformationFailed method.
005     */
006    public interface JsonGetStockInformationListener {
007            /**
008             * 
009             * @param data The method that should be overridden to handle the data if the method was successful.
010             */
011            public abstract void getStockInformationCompleted(String data);
012            /**
013             * 
014             * @param error The method that should be overridden to handle an exception that occurred while getting the SearchResponse.
015             */
016            public abstract void getStockInformationFailed(Exception error);
017    }
